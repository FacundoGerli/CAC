ORG 1000H
cadena db "AAAAAAAA"
finTabla db ?
caracter db "A"

ORG 3000H
CONCAR:MOV CX,BX
SUB CX,AX
DEC BX
MOV Al,0
LAZO:INC BX
CMP CX,0
JZ VOLVER
CMP DX,[BX]
JNZ LAZO
INC AL
MOV BYTE PTR [BX],58h
JMP LAZO


VOLVER:RET
ORG 2000H

MOV AX, OFFSET cadena
MOV BX, OFFSET finTabla
MOV DX,caracter
CALL CONCAR


FIN:HLT
END
